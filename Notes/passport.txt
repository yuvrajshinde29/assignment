------------------ | ------------------------- | -------------------------------------

	step 3:
		configure local strategy
		
		import { Strategy as LocalStrategy } from "passport-local";
		
		passport.use(
			new LocalStrategy({usernameField: 'email', passwordField: 'password'},(email, password, done)=>{
				
				//fetch user from db
				const user = User.findOne({where:{
								email
							}})
				
				if(!user) return done(null,false,{message: 'user not found'});
				
				if(user.password !== password) return done(null, false, {message: 'Invalid password'})
				
				return done(null, user)	//----
			})		
		)
		-----
		startgey tell passport how to authenticate user.
		
		LocalStrategy checks username and password
		and callback where we verify credentials. (email,password, done)
			if valid we call done(null, user) and passport internally call req.logIn(user) or we can do it manually and store user in req.user
			if invalid we call done(null, false, {message:''}) -req end with error or redirect depending on you
			
				done(error | null, object | false, object/string Optional message why auth failed)
		-----
	step 4:
		serialize and deserialize
		
		passport.serializeUser((user, done)=>{
			done(null,user.id)		//store user data in session
		})
		
		
		passport.deserializeUser((id, done)=>{
			done(null, id)
		})
		
		---
		after logged in passport need to remember who is logged in.
		serialize:  save user id in session.
		deserialize: for every req after login passport automatically runs the deserializeUser and store returned data into req.user

	custom callback to handle JSON API responses:
	
	app.post('/login', (req, res, next) => {
		passport.authenticate('local', (err, user, info) => {
			if (err) return next(err);
			if (!user) return res.status(401).json({ message: info.message });
			
			//this created the session
			req.logIn(user, (err) => {
				if (err) return next(err);
				res.json({ message: 'Login successful', user });
			});
		})(req, res, next);
	});

		};
		
		//2: verify callback - run whenever request hit to protected route
		passoprt.use(
			new JWTStartgey(opts, async (jwt_payload, done) => {

					const id = jwt_payload.sub
					const user = await User.findOne({where:{
						id
					}});
				
					if(user) return done(null,user)         //success attch user to req.user
					else return  done(null,false,{error:'unauthorized | invalid user'}) // no user -> unauthorized
				
				} catch (error) {
					console.log(error.messsage);
					resizeBy.status(500).json({ error: error.messsage });
				}
			})
		);
