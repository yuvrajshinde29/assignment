	const { Pool } = require("pg");

	const pool = new Pool({
		user: "postgres",
		host: "localhost",
		database: "postgres",
		password: "Nimap@29",
		port: "5432",
	});
	
	async function connectDB() {
		try {
			const clint = await pool.connect();
			console.log("connected to postgress");
			clint.release(); //release connection to pool
		} catch (error) {
			console.error("connection error : ", error.stack);
		}
	}
	connectDB();
	module.exports = pool;
