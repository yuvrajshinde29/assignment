Nodemailer:
	Simple node.js module that allows you to send emails directly from your server.(SMPT, Gmail, Outlook,etc)
	
	use: 
		password reset links,
		User sign up verification emails
		Notifications or reports
		
	steps:
		1. create transport
		2. defined the email content  
		3. send it 
		-------------
		1. Insatall nodemailer: 
			npm i nodemailer
			
				// config/email.js
				//1. create a transpoter(connect to email service)
				const transporter = nodemailer.createTransport({
					service: "gmail", // or use host, port and secure for custom servers
					auth: {
						user: process.env.USER,
						pass: process.env.PASSOWORD, //not gmail password
					},
				});
				
				//verify connection (optional) 
				transpoter.verify((err,success)=>{
					if(err)return console.log('email server connection failed', err) 
					console.log('email server ready to send message')
				})
				module.exports = transporter;
				
			// utils/sendEmail.js
			async function sendEmail(to, subject,html,text){
			
				//2. define email content
				const mailOptions = {
					from: `"Demo Nodemailer" <${process.env.EMAIL_USER}>`,

					subject,

				};
				
				//3.send the email

	--------------- 
	Multiple receivers:
		to send email to more than one receiver add them to the 'to' property of the mailoptions
		
		let mailOptions = {
			from: 'youremail@gmail.com',
			to: 'firstreciver@gmail.com, secondreceiver@gmail.com,...'
			subject: '',
			text:''
		}
